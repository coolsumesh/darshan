import type { OpenClawPluginApi, PluginRuntime } from "openclaw/plugin-sdk";
import { emptyPluginConfigSchema } from "openclaw/plugin-sdk";

const CHANNEL_ID = "darshan";
const POLL_INTERVAL_MS = 3000;     // main loop tick
const NOTIF_POLL_MS   = 30000;     // notification poll fallback (every 30s)

let _runtime: PluginRuntime | null = null;

function getDarshanConfig(cfg: any) {
  const c = cfg?.channels?.darshan ?? {};
  const baseUrl = (
    c.endpoint ??
    process.env.DARSHAN_BASE_URL ??
    "https://darshan.caringgems.in/api/backend"
  ).replace(/\/$/, "");
  const agentId    = c.agentId    ?? process.env.DARSHAN_AGENT_ID    ?? null;
  const agentToken = c.agentToken ?? process.env.DARSHAN_AGENT_TOKEN ?? null;
  return { baseUrl, agentId, agentToken };
}

// ── Process a single notification ────────────────────────────────────────────
async function processNotification(
  notif: {
    notification_id: string;
    message_id:      string;
    thread_id:       string;
    thread_subject?: string;
    message_body:    string;
    message_from:    string;   // sender_slug
    priority?:       string;
  },
  baseUrl:    string,
  agentId:    string,
  agentToken: string,
  log:        any
) {
  const headers = { Authorization: `Bearer ${agentToken}`, "Content-Type": "application/json" };

  const markProcessed = async (note: string) => {
    await fetch(`${baseUrl}/api/v1/notifications/${notif.notification_id}/process`, {
      method:  "POST",
      headers,
      body:    JSON.stringify({ response_note: note }),
      signal:  AbortSignal.timeout(10000),
    }).catch(() => {});
  };

  const sendReply = async (replyText: string) => {
    const clean = replyText.trim();
    if (!clean || clean === "NO_REPLY" || clean === "HEARTBEAT_OK") return false;
    await fetch(`${baseUrl}/api/v1/threads/${notif.thread_id}/messages`, {
      method:  "POST",
      headers,
      body:    JSON.stringify({ body: clean }),
      signal:  AbortSignal.timeout(15000),
    }).catch(() => {});
    log?.info?.(`[darshan] reply sent in thread ${notif.thread_id.slice(0, 8)}: ${clean.slice(0, 60)}`);
    return true;
  };

  try {
    const rt  = _runtime!;
    const cfg = await (rt as any).config.loadConfig();
    let delivered = false;

    await (rt as any).channel.reply.dispatchReplyWithBufferedBlockDispatcher({
      ctx: {
        Body:               notif.message_body,
        From:               notif.message_from,
        To:                 agentId,
        SessionKey:         `darshan:thread:${notif.thread_id}`,
        AccountId:          "default",
        OriginatingChannel: CHANNEL_ID as any,
        OriginatingTo:      notif.thread_id,
        ChatType:           "direct" as const,
        SenderName:         notif.message_from,
        Subject:            notif.thread_subject ?? "",
      },
      cfg,
      dispatcherOptions: {
        deliver: async (payload: { text?: string; body?: string }) => {
          const out = payload?.text ?? payload?.body ?? "";
          const sent = await sendReply(out);
          if (sent) {
            delivered = true;
            await markProcessed(`replied: ${out.slice(0, 80)}`);
          }
        },
      },
    });

    if (!delivered) {
      await markProcessed("auto-processed: no reply generated");
    }
  } catch (err: any) {
    log?.error?.(`[darshan] notification ${notif.notification_id} error: ${err?.message ?? err}`);
    await markProcessed(`error: ${err?.message ?? "unknown"}`);
  }
}

// ── WS real-time push ─────────────────────────────────────────────────────────
function startNotificationWs(
  wsUrl:       string,
  agentId:     string,
  agentToken:  string,
  onNotif:     (data: any) => void,
  abortSignal: AbortSignal | null,
  log:         any
) {
  if (abortSignal?.aborted) return;

  let ws: any = null;
  let reconnectTimer: ReturnType<typeof setTimeout> | null = null;

  const cleanup = () => {
    if (reconnectTimer) { clearTimeout(reconnectTimer); reconnectTimer = null; }
    try { ws?.close?.(); } catch {}
  };

  abortSignal?.addEventListener?.("abort", cleanup, { once: true });

  const connect = () => {
    if (abortSignal?.aborted) return;
    try {
      const WebSocket = (globalThis as any).WebSocket ?? require("ws");
      ws = new WebSocket(`${wsUrl}?agentId=${agentId}&token=${agentToken}`);

      ws.onopen = () => {
        log?.info?.(`[darshan] WS connected (${wsUrl})`);
      };

      ws.onmessage = (evt: any) => {
        try {
          const msg = JSON.parse(evt.data ?? evt);
          // Event format: { type: "notification", data: { notification_id, thread_id, ... } }
          if (msg?.type === "notification" && msg?.data) {
            onNotif(msg.data);
          }
        } catch {}
      };

      ws.onerror = (err: any) => {
        log?.warn?.(`[darshan] WS error: ${err?.message ?? err}`);
      };

      ws.onclose = () => {
        if (!abortSignal?.aborted) {
          log?.info?.("[darshan] WS closed, reconnecting in 15s");
          reconnectTimer = setTimeout(connect, 15000);
        }
      };
    } catch (err: any) {
      log?.warn?.(`[darshan] WS init error: ${err?.message ?? err}`);
      if (!abortSignal?.aborted) {
        reconnectTimer = setTimeout(connect, 15000);
      }
    }
  };

  connect();
}

const plugin = {
  id:          CHANNEL_ID,
  name:        "Darshan",
  description: "Darshan native channel plugin (notifications + threads)",
  configSchema: emptyPluginConfigSchema(),

  register(api: OpenClawPluginApi) {
    _runtime = (api as any).runtime ?? null;
    console.log("[darshan] register() called, runtime:", !!_runtime);

    api.registerChannel({
      plugin: {
        id: CHANNEL_ID,
        meta: {
          id:             CHANNEL_ID,
          label:          "Darshan",
          selectionLabel: "Darshan",
          detailLabel:    "Darshan",
          blurb:          "Native Darshan channel",
          order:          95,
        },
        capabilities: {
          chatTypes:      ["direct" as const, "group" as const],
          media:          false,
          threads:        true,
          reactions:      false,
          edit:           false,
          unsend:         false,
          reply:          true,
          effects:        false,
          blockStreaming: false,
        },
        reload: { configPrefixes: ["channels.darshan"] },
        configSchema: emptyPluginConfigSchema(),
        config: {
          listAccountIds:     (_cfg: any) => ["default"],
          resolveAccount:     (cfg: any)  => ({
            accountId: "default",
            enabled:   cfg?.channels?.darshan?.enabled !== false,
          }),
          describeAccount:    (_account: any, _cfg: any) => ({ configured: true }),
          isConfigured:       async (_account: any, _cfg: any) => true,
          defaultAccountId:   () => "default",
          setAccountEnabled:  ({ cfg, enabled }: any) => ({
            ...cfg,
            channels: { ...(cfg?.channels ?? {}), darshan: { ...(cfg?.channels?.darshan ?? {}), enabled } },
          }),
        },
        messaging: { normalizeTarget: (t: string) => t?.trim() || undefined },
        directory: { self: async () => null, listPeers: async () => [], listGroups: async () => [] },
        outbound: {
          deliveryMode:  "gateway" as const,
          textChunkLimit: 4000,
          // Proactive send (agent → agent): uses POST /threads/direct
          sendText: async ({ to, text }: any) => {
            const cfg = await (_runtime as any)?.config?.loadConfig?.() ?? {};
            const { baseUrl, agentId, agentToken } = getDarshanConfig(cfg);
            if (agentId && agentToken && text?.trim()) {
              await fetch(`${baseUrl}/api/v1/threads/direct`, {
                method:  "POST",
                headers: {
                  Authorization:  `Bearer ${agentToken}`,
                  "Content-Type": "application/json",
                },
                body:   JSON.stringify({ to, body: text.trim() }),
                signal: AbortSignal.timeout(15000),
              }).catch(() => {});
            }
            return {
              channel:   CHANNEL_ID,
              messageId: `darshan-${Date.now()}`,
              chatId:    String(to ?? "unknown"),
            };
          },
        },
        gateway: {
          startAccount: async (ctx: any) => {
            const { cfg, abortSignal, setStatus, log } = ctx;
            const { baseUrl, agentId, agentToken } = getDarshanConfig(cfg);

            setStatus?.({ configured: true, running: true, lastError: null });
            log?.info?.(`[darshan] started, polling ${baseUrl}`);
            console.log(`[darshan] startAccount called, polling ${baseUrl}`);

            if (!agentId || !agentToken) {
              log?.warn?.("[darshan] no agentId/agentToken — skipping");
              return { stop: () => {} };
            }

            const headers = { Authorization: `Bearer ${agentToken}`, "Content-Type": "application/json" };

            // ── WS real-time push ────────────────────────────────────────────
            const wsBase = baseUrl
              .replace(/^https:\/\//, "wss://")
              .replace(/^http:\/\//, "ws://");

            // Set of notification_ids already being processed (avoid double-process)
            const inFlight = new Set<string>();

            const handleWsNotif = (data: any) => {
              const nid = data?.notification_id;
              if (!nid || inFlight.has(nid)) return;
              inFlight.add(nid);
              log?.info?.(`[darshan] WS push: notification ${nid}`);
              // Fetch full notification to get message body
              fetch(`${baseUrl}/api/v1/notifications?status=pending&limit=50`, { headers })
                .then((r) => r.json())
                .then((d: any) => {
                  const found = (d.notifications ?? []).find((n: any) => n.notification_id === nid);
                  if (found) {
                    processNotification(found, baseUrl, agentId, agentToken, log)
                      .catch(() => {})
                      .finally(() => inFlight.delete(nid));
                  } else {
                    inFlight.delete(nid);
                  }
                })
                .catch(() => inFlight.delete(nid));
            };

            startNotificationWs(
              `${wsBase}/ws`, agentId, agentToken,
              handleWsNotif, abortSignal, log
            );

            // ── Poll fallback (every 30s) ─────────────────────────────────────
            let lastPoll = 0;

            async function pollNotifications() {
              try {
                const res = await fetch(
                  `${baseUrl}/api/v1/notifications?status=pending&limit=50`,
                  { headers: { Authorization: `Bearer ${agentToken}` }, signal: AbortSignal.timeout(10000) }
                );
                if (!res.ok) return;
                const data = await res.json() as { notifications?: any[] };
                const pending = (data.notifications ?? []).filter(
                  (n: any) => !inFlight.has(n.notification_id)
                );
                for (const notif of pending) {
                  inFlight.add(notif.notification_id);
                  processNotification(notif, baseUrl, agentId, agentToken, log)
                    .catch((e: any) => log?.error?.(`[darshan] notif error: ${e?.message ?? e}`))
                    .finally(() => inFlight.delete(notif.notification_id));
                }
              } catch (err: any) {
                log?.warn?.(`[darshan] poll error: ${err?.message ?? err}`);
              }
            }

            // ── Main loop ─────────────────────────────────────────────────────
            while (!abortSignal?.aborted) {
              if (Date.now() - lastPoll > NOTIF_POLL_MS) {
                lastPoll = Date.now();
                pollNotifications().catch(() => {});
              }

              await new Promise<void>((resolve) => {
                if (abortSignal?.aborted) { resolve(); return; }
                const t = setTimeout(resolve, POLL_INTERVAL_MS);
                abortSignal?.addEventListener?.("abort", () => { clearTimeout(t); resolve(); }, { once: true });
              });
            }

            log?.info?.("[darshan] stopped");
            console.log("[darshan] startAccount exiting");
            return { stop: () => {} };
          },
          stopAccount: async () => {},
        },
      },
    });
  },
};

export default plugin;
