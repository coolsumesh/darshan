import type { OpenClawPluginApi, PluginRuntime } from "openclaw/plugin-sdk";
import { emptyPluginConfigSchema } from "openclaw/plugin-sdk";

const CHANNEL_ID = "darshan";
const POLL_INTERVAL_MS = 3000;

let _runtime: PluginRuntime | null = null;
const _processingThreads = new Set<string>(); // concurrency guard per thread

// ── Token usage tracking ──────────────────────────────────────────────────────
import { readFileSync } from "node:fs";
import { homedir } from "node:os";

const _sessionTokenSnapshots = new Map<string, number>();

function readSessionTokens(sessionKey: string): { totalTokens: number; model: string } | null {
  try {
    const sessionsPath = `${homedir()}/.openclaw/agents/main/sessions/sessions.json`;
    const raw = readFileSync(sessionsPath, "utf8");
    const sessions = JSON.parse(raw) as Record<string, { totalTokens?: number; model?: string }>;
    const entry = sessions[sessionKey];
    if (!entry) return null;
    return { totalTokens: entry.totalTokens ?? 0, model: entry.model ?? "unknown" };
  } catch {
    return null;
  }
}

async function recordUsageEvent(
  baseUrl: string,
  agentToken: string,
  sessionKey: string,
  threadId: string,
  agentId: string,
): Promise<void> {
  try {
    const current = readSessionTokens(sessionKey);
    if (!current) return;
    const last = _sessionTokenSnapshots.get(sessionKey) ?? 0;
    const delta = Math.max(0, current.totalTokens - last);
    _sessionTokenSnapshots.set(sessionKey, current.totalTokens);
    if (delta === 0) return;
    await fetch(`${baseUrl}/api/v1/usage`, {
      method: "POST",
      headers: { "Content-Type": "application/json", "Authorization": `Bearer ${agentToken}` },
      body: JSON.stringify({ session_key: sessionKey, thread_id: threadId, agent_id: agentId, model: current.model, tokens_delta: delta, tokens_total: current.totalTokens }),
    });
  } catch { }
}

// ── Responder status emission ─────────────────────────────────────────────────
type ResponderStatus = "queued" | "picked" | "thinking" | "responded" | "blocked" | "failed";

async function emitResponderStatus(
  baseUrl: string,
  agentToken: string,
  threadId: string,
  sourceMessageId: string,
  status: ResponderStatus,
  extra?: { intent?: string; failure_code?: string; failure_message?: string },
  log?: any
): Promise<void> {
  try {
    await fetch(`${baseUrl}/api/v1/threads/${threadId}/reply-status`, {
      method: "POST",
      headers: { Authorization: `Bearer ${agentToken}`, "Content-Type": "application/json" },
      body: JSON.stringify({ source_message_id: sourceMessageId, status, ...extra }),
      signal: AbortSignal.timeout(8000),
    });
  } catch (e: any) {
    log?.warn?.(`[darshan] emitResponderStatus error: ${e?.message}`);
  }
}

// ── Config helper ─────────────────────────────────────────────────────────────
type DarshanNotif = {
  notification_id: string;
  thread_id: string;
  message_id: string;
  recipient_id: string;
  message_body?: string;
  message_from?: string;
  thread_subject?: string;
  priority?: string;
  sender_is_agent?: boolean;
};

function getDarshanConfig(cfg: any) {
  const c = cfg?.channels?.darshan ?? {};
  const baseUrl = (c.endpoint ?? process.env.DARSHAN_BASE_URL ?? "https://darshan.caringgems.in/api/backend").replace(/\/$/, "");
  const apiKey = c.apiKey ?? process.env.DARSHAN_API_KEY ?? "";
  const agentId = c.agentId ?? process.env.DARSHAN_AGENT_ID ?? null;
  const agentToken = c.agentToken ?? process.env.DARSHAN_AGENT_TOKEN ?? null;
  // deliverySource: "redis" = WS-bridged (default), "notifications" = legacy polling fallback
  const deliverySource: "redis" | "notifications" = c.deliverySource ?? "redis";
  return { baseUrl, apiKey, agentId, agentToken, deliverySource };
}

// ── Core notification handler (shared by both paths) ──────────────────────────
async function handleNotification(
  notif: DarshanNotif & { source_message_id?: string },
  baseUrl: string,
  _agentId: string,
  agentToken: string,
  log: any
) {
  let fromSlug = notif.message_from ?? "";
  let body = notif.message_body ?? "";
  const threadId = notif.thread_id;
  const sourceMessageId = notif.source_message_id ?? notif.message_id;

  if ((!body || !fromSlug) && notif.message_id && threadId) {
    try {
      const msgRes = await fetch(`${baseUrl}/api/v1/threads/${threadId}/messages/${notif.message_id}`, {
        headers: { Authorization: `Bearer ${agentToken}` },
        signal: AbortSignal.timeout(8000),
      });
      if (msgRes.ok) {
        const msgData: any = await msgRes.json();
        const msg = msgData.message ?? msgData;
        if (!body) body = msg.body ?? "";
        if (!fromSlug) fromSlug = msg.sender_slug ?? msg.sender_name ?? "agent";
      }
    } catch { }
  }
  if (!fromSlug) fromSlug = "agent";

  const postReply = async (replyText: string): Promise<boolean> => {
    const clean = replyText.trim();
    if (!clean || clean === "NO_REPLY" || clean === "HEARTBEAT_OK") return false;
    const res = await fetch(`${baseUrl}/api/v1/threads/${threadId}/messages`, {
      method: "POST",
      headers: { Authorization: `Bearer ${agentToken}`, "Content-Type": "application/json" },
      body: JSON.stringify({ body: clean, intents: ["response"], awaiting_on: "none" }),
      signal: AbortSignal.timeout(15000),
    }).catch(() => null);
    if (!res?.ok) return false;
    log?.warn?.(`[darshan] replied in thread: ${clean.slice(0, 60)}`);
    return true;
  };

  if (notif.sender_is_agent) {
    log?.info?.(`[darshan] skipping auto-reply — sender ${fromSlug} is an agent`);
    return;
  }

  try {
    // picked: agent has claimed this event
    await emitResponderStatus(baseUrl, agentToken, threadId, sourceMessageId, "picked", undefined, log);

    const rt = _runtime!;
    const cfg = await rt.config.loadConfig();
    const { agentId: aId } = getDarshanConfig(cfg);
    const route = rt.channel.routing.resolveAgentRoute({
      cfg, channel: CHANNEL_ID, accountId: "default",
      peer: { kind: "direct", id: fromSlug },
    });
    const threadSessionKey = route.sessionKey.replace(/direct:[^:]+$/, `thread:${threadId}`);

    let delivered = false;

    await rt.channel.reply.dispatchReplyWithBufferedBlockDispatcher({
      ctx: {
        Body: body,
        BodyForAgent: `[Darshan thread: "${notif.thread_subject ?? "no subject"}" | thread_id: ${threadId}]\n${fromSlug}: ${body}`,
        From: fromSlug,
        To: "Sanjaya",
        SessionKey: threadSessionKey,
        AccountId: route.accountId,
        ChatType: "direct",
        SenderName: fromSlug,
        SenderId: fromSlug,
        Surface: CHANNEL_ID,
        Provider: CHANNEL_ID,
        OriginatingTo: fromSlug,
      },
      cfg,
      dispatcherOptions: {
        onFirstChunk: async () => {
          // thinking: LLM started generating
          await emitResponderStatus(baseUrl, agentToken, threadId, sourceMessageId, "thinking", undefined, log);
        },
        deliver: async (payload: { text?: string }) => {
          const out = (payload?.text ?? "").trim();
          const ok = await postReply(out);
          if (ok) {
            delivered = true;
            await emitResponderStatus(baseUrl, agentToken, threadId, sourceMessageId, "responded", { intent: "response" }, log);
            log?.warn?.("[darshan] real reply posted");
            const dCfg = getDarshanConfig(cfg);
            void recordUsageEvent(dCfg.baseUrl, dCfg.agentToken ?? "", threadSessionKey, threadId, aId ?? "");
          }
        },
        onSkip: (_payload: any, info: { kind: string; reason: string }) => {
          log?.warn?.(`[darshan] reply skipped kind=${info.kind} reason=${info.reason}`);
        },
        onError: async (e: unknown, info: { kind: string }) => {
          log?.warn?.(`[darshan] reply dispatch error kind=${info.kind}: ${String((e as any)?.message ?? e)}`);
          await emitResponderStatus(baseUrl, agentToken, threadId, sourceMessageId, "failed", {
            failure_code: info.kind,
            failure_message: String((e as any)?.message ?? "dispatch error"),
          }, log);
        },
      },
    });

    if (!delivered) {
      log?.warn?.(`[darshan] dispatch returned without immediate delivery — reply queued`);
    }
  } catch (err: any) {
    log?.error?.(`[darshan] handleNotification error: ${err?.message}`);
    await emitResponderStatus(baseUrl, agentToken, threadId, sourceMessageId, "failed", {
      failure_code: "exception",
      failure_message: String(err?.message ?? err).slice(0, 200),
    }, log);
  }
}

// ── WebSocket helper ──────────────────────────────────────────────────────────
let _wsReconnectTimer: ReturnType<typeof setTimeout> | null = null;

function startAgentWs(
  wsUrl: string,
  agentId: string,
  agentToken: string,
  onEvent: (type: string, data: any) => void,
  abortSignal: AbortSignal,
  log: any
) {
  if (abortSignal.aborted) return;
  let ws: any;
  try {
    ws = new (globalThis as any).WebSocket(wsUrl);
  } catch {
    log?.warn?.("[darshan] WebSocket not available");
    return;
  }

  ws.onopen = () => {
    log?.info?.("[darshan] WS connected");
    ws.send(JSON.stringify({ type: "agent_auth", agent_id: agentId, token: agentToken }));
    const pingInterval = setInterval(() => {
      if (ws.readyState === 1) {
        try { ws.send(JSON.stringify({ type: "ping" })); } catch { }
      } else {
        clearInterval(pingInterval);
      }
    }, 30000);
    abortSignal.addEventListener("abort", () => clearInterval(pingInterval), { once: true });
  };

  ws.onmessage = (event: any) => {
    try {
      const msg = JSON.parse(typeof event.data === "string" ? event.data : event.data.toString());
      if (msg.type === "agent_subscribed") {
        log?.info?.(`[darshan] WS subscribed for agent ${agentId}`);
      } else {
        onEvent(msg.type, msg.data ?? msg);
      }
    } catch { }
  };

  ws.onclose = () => {
    if (abortSignal.aborted) return;
    log?.warn?.("[darshan] WS closed, reconnecting in 5s...");
    _wsReconnectTimer = setTimeout(() => {
      startAgentWs(wsUrl, agentId, agentToken, onEvent, abortSignal, log);
    }, 5000);
  };

  ws.onerror = () => log?.warn?.("[darshan] WS error");
  abortSignal.addEventListener("abort", () => {
    if (_wsReconnectTimer) clearTimeout(_wsReconnectTimer);
    try { ws.close(); } catch { }
  }, { once: true });
}

async function pollNotifications(
  baseUrl: string,
  agentToken: string,
  onNotif: (notif: any) => void,
  log: any
) {
  try {
    const res = await fetch(`${baseUrl}/api/v1/notifications?status=pending&limit=50`, {
      headers: { Authorization: `Bearer ${agentToken}` },
      signal: AbortSignal.timeout(10000),
    });
    if (!res.ok) return;
    const data: any = await res.json();
    for (const notif of data.notifications ?? []) onNotif(notif);
  } catch (err: any) {
    log?.warn?.(`[darshan] notification poll error: ${err?.message ?? err}`);
  }
}

// ── Plugin definition ─────────────────────────────────────────────────────────
const plugin = {
  id: CHANNEL_ID,
  name: "Darshan",
  description: "Darshan native channel plugin",
  configSchema: emptyPluginConfigSchema(),

  register(api: OpenClawPluginApi) {
    _runtime = api.runtime;

    api.registerChannel({
      plugin: {
        id: CHANNEL_ID,
        meta: {
          id: CHANNEL_ID,
          label: "Darshan",
          selectionLabel: "Darshan",
          detailLabel: "Darshan",
          blurb: "Native Darshan channel",
          order: 95,
        },
        capabilities: {
          chatTypes: ["direct" as const, "group" as const],
          media: false, threads: true, reactions: false,
          edit: false, unsend: false, reply: true, effects: false, blockStreaming: false,
        },
        reload: { configPrefixes: ["channels.darshan"] },
        configSchema: emptyPluginConfigSchema(),
        config: {
          listAccountIds: (_cfg: any) => ["default"],
          resolveAccount: (cfg: any) => ({ accountId: "default", enabled: cfg?.channels?.darshan?.enabled !== false }),
          describeAccount: (_a: any, _c: any) => ({ configured: true }),
          isConfigured: async () => true,
          defaultAccountId: () => "default",
          setAccountEnabled: ({ cfg, enabled }: any) => ({
            ...cfg,
            channels: { ...(cfg?.channels ?? {}), darshan: { ...(cfg?.channels?.darshan ?? {}), enabled } },
          }),
        },
        messaging: { normalizeTarget: (t: string) => t?.trim() || undefined },
        directory: {
          self: async () => {
            const cfg = _runtime ? await _runtime.config.loadConfig() : {} as any;
            const { agentId } = getDarshanConfig(cfg);
            return agentId ? { kind: "user" as const, id: agentId, name: "Sanjaya" } : null;
          },
          listPeers: async () => {
            const cfg = _runtime ? await _runtime.config.loadConfig() : {} as any;
            const { baseUrl, agentId, agentToken } = getDarshanConfig(cfg);
            if (!agentId || !agentToken) return [];
            try {
              const projectId = "702072b8-3264-4a9e-9827-aec2eba1d686";
              const res = await fetch(`${baseUrl}/api/v1/projects/${projectId}/agents`, {
                headers: { Authorization: `Bearer ${agentToken}` },
                signal: AbortSignal.timeout(8000),
              });
              if (!res.ok) return [];
              const data = await res.json() as { agents?: any[] };
              return (data.agents ?? [])
                .filter((a: any) => a.id !== agentId)
                .map((a: any) => ({ kind: "user" as const, id: a.id, name: a.name ?? a.slug }));
            } catch { return []; }
          },
          listGroups: async () => [],
        },
        outbound: {
          deliveryMode: "gateway" as const,
          textChunkLimit: 4000,
          sendText: async ({ to, text }: any) => {
            const cfg = _runtime ? await _runtime.config.loadConfig() : {} as any;
            const { baseUrl, agentId, agentToken } = getDarshanConfig(cfg);
            if (agentId && agentToken && text) {
              await fetch(`${baseUrl}/api/v1/threads/direct`, {
                method: "POST",
                headers: { Authorization: `Bearer ${agentToken}`, "Content-Type": "application/json" },
                body: JSON.stringify({ to, body: text }),
                signal: AbortSignal.timeout(15000),
              }).catch(() => { });
            }
            return { channel: CHANNEL_ID, messageId: `darshan-${Date.now()}`, chatId: String(to ?? "unknown") };
          },
          // Required by OpenClaw delivery handler (checked alongside sendText).
          // Darshan has media:false so this is a stub — media messages are not supported.
          sendMedia: async ({ to }: any) => {
            return { channel: CHANNEL_ID, messageId: `darshan-${Date.now()}`, chatId: String(to ?? "unknown") };
          },
        },
        gateway: {
          startAccount: async (ctx: any) => {
            const { cfg, abortSignal, setStatus, log } = ctx;
            const { baseUrl, agentId, agentToken, deliverySource } = getDarshanConfig(cfg);

            setStatus?.({ configured: true, running: true, lastError: null });
            log?.info?.(`[darshan] started — delivery=${deliverySource} platform=${process.platform}`);

            const processedNotifIds = new Set<string>();

            const handleIncomingNotif = (notif: any) => {
              if (!notif?.notification_id) return;
              if (processedNotifIds.has(notif.notification_id)) return;
              processedNotifIds.add(notif.notification_id);
              if (processedNotifIds.size > 500) {
                const first = processedNotifIds.values().next().value;
                if (first) processedNotifIds.delete(first);
              }
              const threadId = notif.thread_id ?? "unknown";
              if (_processingThreads.has(threadId)) {
                log?.warn?.(`[darshan] skipping notif ${notif.notification_id} - thread ${threadId} already processing`);
                return;
              }
              _processingThreads.add(threadId);
              log?.info?.(`[darshan] notification from ${notif.message_from ?? "?"}: ${String(notif.message_body ?? "").slice(0, 60)}`);
              handleNotification(notif, baseUrl, agentId!, agentToken!, log)
                .catch((e: any) => log?.error?.(`[darshan] handleNotification error: ${e?.message ?? e}`))
                .finally(() => _processingThreads.delete(threadId));
            };

            // Start WS — handles both reply_required events and legacy notifications
            if (agentId && agentToken) {
              const wsUrl = baseUrl.replace(/^https:\/\//, "wss://").replace(/^http:\/\//, "ws://") + "/ws";
              startAgentWs(wsUrl, agentId, agentToken, async (type: string, data: any) => {

                if (type === "reply_required") {
                  // Primary path: server pushed a reply_required event via WS (bridged from Redis stream)
                  const threadId = data?.thread_id ?? "unknown";
                  if (_processingThreads.has(threadId)) {
                    log?.warn?.(`[darshan] reply_required: thread ${threadId} already processing — skipping`);
                    return;
                  }
                  _processingThreads.add(threadId);
                  log?.info?.(`[darshan] reply_required for thread ${threadId} msg ${data?.message_id}`);
                  const notif = {
                    notification_id: data?.event_id ?? `rr-${Date.now()}`,
                    thread_id: threadId,
                    message_id: data?.message_id,
                    source_message_id: data?.message_id,
                    recipient_id: agentId!,
                    message_body: "",
                    message_from: "",
                    thread_subject: "",
                    sender_is_agent: false,
                  };
                  handleNotification(notif, baseUrl, agentId!, agentToken!, log)
                    .catch((e: any) => log?.error?.(`[darshan] reply_required handler error: ${e?.message}`))
                    .finally(() => _processingThreads.delete(threadId));

                } else if (type === "notification" && deliverySource === "notifications") {
                  // Fallback path: legacy notification-based delivery
                  handleIncomingNotif(data);

                } else if (type === "ping") {
                  fetch(`${baseUrl}/api/v1/agents/${agentId}/pong`, {
                    method: "POST",
                    headers: { "Content-Type": "application/json" },
                    body: JSON.stringify({ token: agentToken, sent_at: data?.sent_at }),
                    signal: AbortSignal.timeout(10000),
                  }).catch(() => { });
                }
              }, abortSignal, log);
            }

            // Notification polling fallback (when deliverySource === "notifications")
            let lastNotifPoll = 0;
            const NOTIF_POLL_MS = 30000;

            while (!abortSignal?.aborted) {
              if (deliverySource === "notifications" && agentId && agentToken && Date.now() - lastNotifPoll > NOTIF_POLL_MS) {
                lastNotifPoll = Date.now();
                pollNotifications(baseUrl, agentToken, handleIncomingNotif, log).catch(() => { });
              }
              await new Promise<void>((resolve) => {
                if (abortSignal?.aborted) { resolve(); return; }
                const t = setTimeout(resolve, POLL_INTERVAL_MS);
                abortSignal?.addEventListener?.("abort", () => { clearTimeout(t); resolve(); }, { once: true });
              });
            }

            log?.info?.("[darshan] stopped");
            return { stop: () => { } };
          },
          stopAccount: async () => { },
        },
      },
    });
  },
};

export default plugin;
